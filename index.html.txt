  (function() {
    // WhatsApp integration for contact form + pre-filled message
    const phoneNumber = "9182966832";  // official number from the brief (9182966832)
    const sendBtn = document.getElementById('sendWhatsappBtn');
    const nameInput = document.getElementById('userName');
    const phoneInput = document.getElementById('userPhone');
    const problemInput = document.getElementById('userProblem');

    function formatWhatsAppMessage() {
      const name = nameInput ? nameInput.value.trim() : '';
      const userPhone = phoneInput ? phoneInput.value.trim() : '';
      const problem = problemInput ? problemInput.value.trim() : '';

      let message = "Hello Cool Breeze! I need appliance service.%0A";
      if (name) message += `*Name:* ${name}%0A`;
      if (userPhone) message += `*My contact:* ${userPhone}%0A`;
      if (problem) message += `*Issue:* ${problem}%0A`;
      else message += `*Issue:* Need service for AC / appliance.%0A`;
      message += `%0A📍 Location: North Lalaguda / Secunderabad area`;
      return message;
    }

    if (sendBtn) {
      sendBtn.addEventListener('click', (e) => {
        e.preventDefault();
        // simple validation: at least name or phone should be filled
        const nameVal = nameInput ? nameInput.value.trim() : '';
        const phoneVal = phoneInput ? phoneInput.value.trim() : '';
        if (!nameVal && !phoneVal) {
          alert("⚠️ Please enter at least your name or phone number so we can assist you.");
          return;
        }
        const waMessage = formatWhatsAppMessage();
        const waLink = `https://wa.me/${phoneNumber}?text=${waMessage}`;
        window.open(waLink, '_blank');
      });
    }

    // Also make all "Book on WhatsApp" buttons and hero button work dynamically with pre-filled message
    const waButtons = document.querySelectorAll('.btn-wa');
    waButtons.forEach(btn => {
      // avoid double attaching to the main send button, but safe to attach generic fallback
      if (btn.id !== 'sendWhatsappBtn') {
        btn.addEventListener('click', (e) => {
          e.preventDefault();
          let prefillMsg = "Hi Cool Breeze! I need AC or appliance service. Please share details.";
          const customMsg = encodeURIComponent(prefillMsg);
          window.open(`https://wa.me/${phoneNumber}?text=${customMsg}`, '_blank');
        });
      }
    });
  })();
